\begin{algorithm}
            \caption{Algorithmic Design of ACO to Solve a TSP}
            \begin{algorithmic}
            \ENSURE All cities are visited, Route forms Hamiltonian Cycle
            \INPUT Heuristics
            \OUTPUT best path, best path length, route plot in 3D graph
            
            \end{algorithmic}
            \begin{algorithmic}
            
            \PROCEDURE{FindSolution}{} 
                \STATE Initialize necessary parameters and pheromone trials
                \RETURN $distance$, $cost$ and $isValidTimeWindow$
                \WHILE{stopping criterion not satisfied}
                \STATE position each ant in a random starting node
                \STATE generate ant population
                \REPEAT
                \FOR {each ant}
                \STATE choose next node by applying the state transition rule
                \STATE apply dynamic pheromone update
                \ENDFOR \STATE {\bfseries end for}
                \UNTIL {each ant has found a solution}
                \STATE Calculate fitness values associated with each ant
                \STATE Find best solution through selection methods
                \STATE Update best solution
                \STATE Update pheromone trial
                \ENDWHILE \STATE {\bfseries end while}
                
            \ENDPROCEDURE \STATE \bfseries end procedure
            \end{algorithmic}
            \end{algorithm}
